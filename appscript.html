const SHEET_NAME = 'Data'; // Sesuaikan dengan nama Sheet pangkalan data Anda const ADMIN_EMAIL = 'admin.ga@perusahaananda.com'; // Email penerima notifikasi GA const CALENDAR_ID = 'primary'; // Menggunakan kalender utama akun pendeploy script const TIMEZONE = 'Asia/Jakarta'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const dataRange = sheet.getDataRange().getValues(); const headers = dataRange[0]; const idIndex = headers.findIndex(h => h.toLowerCase() === 'id'); const emailIndex = headers.findIndex(h => h.toLowerCase() === 'email'); const otpIndex = headers.findIndex(h => h.toLowerCase() === 'otp'); const eventIndex = headers.findIndex(h => h.toLowerCase() === 'eventid'); // ACTION: ADD if (action === 'add') { let eventId = ""; if (data.type !== 'block') { const eventTitle = `[Mobil] ${data.car} - ${data.name} (${data.divisi})`; const startTime = new Date(`${data.date}T${data.start}:00+07:00`); const endTime = new Date(`${data.date}T${data.end}:00+07:00`); const description = `Tujuan: ${data.tujuan}\nNo WA: ${data.wa}\nEmail: ${data.email}`; const event = CalendarApp.getCalendarById(CALENDAR_ID).createEvent(eventTitle, startTime, endTime, { description: description }); eventId = event.getId(); } const newRow = []; for (let i = 0; i < headers.length; i++) { const header = headers[i].toLowerCase(); if (header === 'id') newRow.push(data.id); else if (header === 'name' || header === 'nama') newRow.push(data.name); else if (header === 'divisi') newRow.push(data.divisi); else if (header === 'wa' || header === 'no wa') newRow.push(data.wa); else if (header === 'email') newRow.push(data.email); else if (header === 'car' || header === 'mobil') newRow.push(data.car); else if (header === 'tujuan') newRow.push(data.tujuan); else if (header === 'date' || header === 'tanggal') newRow.push(data.date); else if (header === 'start' || header === 'jam mulai') newRow.push(data.start); else if (header === 'end' || header === 'jam selesai') newRow.push(data.end); else if (header === 'type' || header === 'tipe') newRow.push(data.type); else if (header === 'eventid') newRow.push(eventId); else if (header === 'otp') newRow.push(""); else newRow.push(""); } sheet.appendRow(newRow); if (data.type !== 'block' && data.email && data.email !== "-") { MailApp.sendEmail({ to: data.email, subject: `Konfirmasi Peminjaman Mobil: ${data.car}`, body: `Halo ${data.name},\n\nPemesanan mobil Anda telah berhasil dicatat:\nMobil: ${data.car}\nTanggal: ${data.date}\nWaktu: ${data.start} - ${data.end} WIB\n\nTerima kasih,\nTim General Affairs` }); MailApp.sendEmail({ to: ADMIN_EMAIL, subject: `Pemesanan Baru - ${data.car}`, body: `Pemesanan mobil:\nNama: ${data.name}\nMobil: ${data.car}\nTanggal: ${data.date}\nWaktu: ${data.start} - ${data.end}\nTujuan: ${data.tujuan}` }); } return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } // ACTION: REQUEST OTP (Otomatis ambil email dari Sheet) if (action === 'requestOTP') { const idStr = String(data.id); for (let i = 1; i < dataRange.length; i++) { if (String(dataRange[i][idIndex]) === idStr) { const userEmail = dataRange[i][emailIndex]; if(userEmail && userEmail !== "-" && userEmail.includes("@")) { const otp = Math.floor(100000 + Math.random() * 900000).toString(); sheet.getRange(i + 1, otpIndex + 1).setValue(otp); MailApp.sendEmail({ to: userEmail, subject: "Kode OTP Sistem Peminjaman Mobil", body: `Kode OTP Anda adalah: ${otp}\nMasukkan kode ini untuk melanjutkan perubahan atau penghapusan pesanan Anda.` }); // Sensor email untuk keamanan saat dikirim ke Frontend let masked = userEmail.replace(/(.{2})(.*)(?=@)/, (gp1, gp2, gp3) => gp2 + "*".repeat(gp3.length)); return ContentService.createTextOutput(JSON.stringify({ status: "success", email: masked })).setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Email tidak valid pada data ini!" })).setMimeType(ContentService.MimeType.JSON); } } } return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Pesanan tidak ditemukan!" })).setMimeType(ContentService.MimeType.JSON); } // ACTION: VERIFY OTP (Untuk mode Edit sebelum memuat form) if (action === 'verifyOTP') { const idStr = String(data.id); for (let i = 1; i < dataRange.length; i++) { if (String(dataRange[i][idIndex]) === idStr && String(dataRange[i][otpIndex]) === String(data.otp)) { return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "OTP Salah!" })).setMimeType(ContentService.MimeType.JSON); } // ACTION: DELETE (Dengan Verifikasi OTP / Admin) if (action === 'delete') { const idStr = String(data.id); const isOtpValid = (data.otp === "admin"); for (let i = 1; i < dataRange.length; i++) { if (String(dataRange[i][idIndex]) === idStr) { if (isOtpValid || String(dataRange[i][otpIndex]) === String(data.otp)) { const oldEventId = dataRange[i][eventIndex]; if (oldEventId) { try { CalendarApp.getCalendarById(CALENDAR_ID).getEventById(oldEventId).deleteEvent(); } catch(e) {} } sheet.deleteRow(i + 1); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "OTP Salah!" })).setMimeType(ContentService.MimeType.JSON); } } } } // ACTION: EDIT (Dengan Verifikasi OTP / Admin) if (action === 'edit') { const idStr = String(data.id); const isOtpValid = (data.otp === "admin"); let targetRowIndex = -1; for (let i = 1; i < dataRange.length; i++) { if (String(dataRange[i][idIndex]) === idStr) { if (isOtpValid || String(dataRange[i][otpIndex]) === String(data.otp)) { targetRowIndex = i; break; } } } if(targetRowIndex !== -1) { // Update Kalender const oldEventId = dataRange[targetRowIndex][eventIndex]; if (oldEventId) { try { CalendarApp.getCalendarById(CALENDAR_ID).getEventById(oldEventId).deleteEvent(); } catch(e) {} } let newEventId = ""; if (data.type !== 'block') { const eventTitle = `[Mobil] ${data.car} - ${data.name} (${data.divisi})`; const startTime = new Date(`${data.date}T${data.start}:00+07:00`); const endTime = new Date(`${data.date}T${data.end}:00+07:00`); const event = CalendarApp.getCalendarById(CALENDAR_ID).createEvent(eventTitle, startTime, endTime, { description: `Tujuan: ${data.tujuan}\nNo WA: ${data.wa}\nEmail: ${data.email}` }); newEventId = event.getId(); } // Memetakan baris yang diperbarui const updatedRow = []; for (let i = 0; i < headers.length; i++) { const header = headers[i].toLowerCase(); if (header === 'name' || header === 'nama') updatedRow.push(data.name); else if (header === 'divisi') updatedRow.push(data.divisi); else if (header === 'wa' || header === 'no wa') updatedRow.push(data.wa); else if (header === 'email') updatedRow.push(data.email); else if (header === 'car' || header === 'mobil') updatedRow.push(data.car); else if (header === 'tujuan') updatedRow.push(data.tujuan); else if (header === 'date' || header === 'tanggal') updatedRow.push(data.date); else if (header === 'start' || header === 'jam mulai') updatedRow.push(data.start); else if (header === 'end' || header === 'jam selesai') updatedRow.push(data.end); else if (header === 'eventid') updatedRow.push(newEventId); else if (header === 'otp') updatedRow.push(""); // Reset OTP setelah edit else updatedRow.push(dataRange[targetRowIndex][i]); // Pertahankan data lain } sheet.getRange(targetRowIndex + 1, 1, 1, headers.length).setValues([updatedRow]); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Data tidak ditemukan atau OTP salah." })).setMimeType(ContentService.MimeType.JSON); } } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.message })).setMimeType(ContentService.MimeType.JSON); } } function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const dataRange = sheet.getDataRange().getValues(); const headers = dataRange[0]; const Peminjamans = []; for (let i = 1; i < dataRange.length; i++) { const row = dataRange[i]; let obj = {}; for (let j = 0; j < headers.length; j++) obj[headers[j].toLowerCase()] = row[j]; Peminjamans.push(obj); } const sheetKaryawan = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Karyawan"); let karyawan = []; if (sheetKaryawan) { const karData = sheetKaryawan.getDataRange().getValues(); for (let i = 1; i < karData.length; i++) { let kObj = {}; for (let j = 0; j < karData[0].length; j++) kObj[karData[0][j].toLowerCase()] = karData[i][j]; karyawan.push(kObj); } } return ContentService.createTextOutput(JSON.stringify({ Peminjamans: Peminjamans, karyawan: karyawan })).setMimeType(ContentService.MimeType.JSON); }
